\begin{frame}
  \frametitle{BIDE model}
  Abundance at time $t$
  \[
    N_t = N_{t-1} + B_{t-1} + I_{t-1} - D_{t-1} - E_{t-1}
  \]
\end{frame}

\begin{frame}
  \frametitle{Geometric growth}
  Abundance at time $t$
  \[
    N_t = N_0(1+r)^t
  \]
  \pause
  \vfill
  Recursive version
  \[
    N_t = N_{t-1} + N_{t-1}r
  \]
\end{frame}
